# 题目
编写一个程序，通过填充空格来解决数独问题。

数独的解法需 遵循如下规则：

1. 数字 1-9 在每一行只能出现一次。
2. 数字 1-9 在每一列只能出现一次。
3. 数字 1-9 在每一个以粗实线分隔的 3x3 宫内只能出现一次。


数独部分空格内已填入了数字，空白格用 '.' 表示。

 

示例：

```
输入：board = 
[["5","3",".",".","7",".",".",".","."],
["6",".",".","1","9","5",".",".","."],
[".","9","8",".",".",".",".","6","."],
["8",".",".",".","6",".",".",".","3"],
["4",".",".","8",".","3",".",".","1"],
["7",".",".",".","2",".",".",".","6"],
[".","6",".",".",".",".","2","8","."],
[".",".",".","4","1","9",".",".","5"],
[".",".",".",".","8",".",".","7","9"]]
输出：
[["5","3","4","6","7","8","9","1","2"],
["6","7","2","1","9","5","3","4","8"],
["1","9","8","3","4","2","5","6","7"],
["8","5","9","7","6","1","4","2","3"],
["4","2","6","8","5","3","7","9","1"],
["7","1","3","9","2","4","8","5","6"],
["9","6","1","5","3","7","2","8","4"],
["2","8","7","4","1","9","6","3","5"],
["3","4","5","2","8","6","1","7","9"]]
```

# 代码
```c++
class Solution {
public:
    bool row[10][10], col[10][10], st[3][3][10];

    bool dfs(int x,int y,vector<vector<char>>& board)
    {
        if(x == 9) return true;
        if(y == 9) {
            return dfs(x + 1, 0, board);
        }
        if(board[x][y] != '.')return dfs(x, y + 1, board);
        else{
            for(int i = 1;i <= 9;i ++)
            {
                if(row[x][i] || col[y][i] || st[x / 3][y / 3][i]) continue;
                row[x][i] = col[y][i] = st[x / 3][y / 3][i] = true;
                board[x][y] = i + '0';
                if (dfs(x, y + 1, board)) return true;
                row[x][i] = col[y][i] = st[x / 3][y / 3][i] = false;
                board[x][y] = '.';
            }
        }
        return false;
    }

    void solveSudoku(vector<vector<char>>& board) {
        for(int i = 0;i < 9;i ++)
        {
            for(int j = 0;j < 9;j ++)
            {
                if(board[i][j] == '.') continue;
                col[j][board[i][j] - '0'] = row[i][board[i][j] - '0'] = true;
                st[i / 3][j / 3][board[i][j] - '0'] = true;
            }
        }
        dfs(0, 0, board);
    }
};
```
